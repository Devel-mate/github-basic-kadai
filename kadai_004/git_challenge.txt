# git_challenge ローカルリポジトリの作成
cd Desktop
mkdir git_challenge
cd git_challenge
git init


# hello_world.txtのファイル作成と日本語訳を追記
echo "Hello, world!" >> hello_world.txt
git add hello_world.txt
git commit -m "hello_world.txtを追加"

echo "こんにちは、世界！" >> hello_world.txt
add hello_world.txt
git commit -m "日本語訳を追記"


# コミット後の変更履歴を表示
git log --oneline
